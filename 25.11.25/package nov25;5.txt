package nov25;
public class StringBuilderExample {

    public static void main(String[] args) {

        String[] arr = { "a", "b", "c" };

        String result = joinWithStringBuilder(arr, ", ");

        System.out.println("Joined String (StringBuilder): " + result);
    }

    public static String joinWithStringBuilder(String[] array, String separator) {
        StringBuilder sb = new StringBuilder();

        for (int i = 0; i < array.length; i++) {
            sb.append(array[i]);

            if (i < array.length - 1) {
                sb.append(separator);
            }
        }

        return sb.toString();
    }
}
